\begin{questions}
\question
Consider the function 
$\map{f}{\matspace_{\nbyn{2}}}{\R^2}$
giving this association.
\begin{equation*}
  f(\begin{mat}
    a  &b  \\
    c  &d
  \end{mat})
  =
  \colvec{a+2c \\ b-d}
\end{equation*}
\begin{parts}
\item What is the dimension of the domain space?
The codomain space?
\item Show that the map preserves addition and scalar multiplication.
\part Show that the map preserves linear combinations.
\item Show that the map is not one-to-one.
\end{parts}


\question
Consider this map $\map{h}{\R^2}{\R^4}$
\begin{equation*}
  \colvec{x \\ y }
  \mapsto
  \colvec{x-y \\ x \\ 0 \\ x+y}
\end{equation*}
\begin{parts}
\part What are the dimensions?
\part Show that the map is linear.
\part Show that the map is not onto.
\end{parts}

\end{questions}

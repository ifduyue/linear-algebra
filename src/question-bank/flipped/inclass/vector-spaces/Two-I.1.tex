\question
Consider this subset of $\R^3$.
\begin{equation*}
  S_1=\set{\colvec{x \\ y \\ z}\suchthat x-y-z=0}
\end{equation*}
\begin{parts}
\part
List three elements and three non-elements.
\part
Show that the set is closed under addition
(with the standard operation).
\part
Show that it is closed under scalar multiplication.
\end{parts}
\begin{solution}
\begin{parts}
\part Three elements are
\begin{equation*}
  \colvec{2 \\ 1 \\ 1}
  \quad
  \colvec{-1 \\ 0 \\ -1}
  \quad
  \colvec{5 \\ 5 \\ 0}
\end{equation*}
and here are three non-elements.
\begin{equation*}
  \colvec{2 \\ 1 \\ 0}
  \quad
  \colvec{-1 \\ 0 \\ 1}
  \quad
  \colvec{0 \\ 5 \\ 5}
\end{equation*}
\part 
Fix $\vec{v},\vec{w}\in S$, that is,
\begin{equation*}
  \colvec{v}=\colvec{v_1 \\ v_2 \\ v_3}
  \qquad
  \colvec{w}=\colvec{w_1 \\ w_2 \\ w_3}
\end{equation*}
such that the first entry minus the second and minus the third is
$v_1-v_2-v_3=0$ and $w_1-w_2-w_3=0$.
Then their sum
\begin{equation*}
  \vec{v}+\vec{w}
  =\colvec{v_1+w_1 \\ v_2+w_2 \\ v_3+w_3}
\end{equation*}
has the property that the first entry minus the second and minus the 
third is 
$(v_1+w_1)-(v_2+w_2)-(v_3+w_3)
=(v_1-v_2-v_3)+(w_1-w_2-w_3)=0$.
\end{parts}
\end{solution}

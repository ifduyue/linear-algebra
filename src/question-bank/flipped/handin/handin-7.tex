\begin{questions}
\question
Multiply the two matrices, or state ``not defined.''
\begin{parts}
\part
\begin{equation*}
\begin{mat}
  4  &3  &1  \\
  2  &0  &5
\end{mat}
\begin{mat}
  -1  &4  &3 \\
  0   &0   &5
\end{mat}
\end{equation*}
\begin{solution}
The product of a $\nbym{2}{3}$ matrix and a $\nbym{2}{3}$ matrix is not defined.
\end{solution}

\part
\begin{equation*}
\begin{mat}
  4  &3  &1  \\
  2  &0  &5
\end{mat}
\begin{mat}
  -1  &0  \\
  4   &0  \\
  3   &5
\end{mat}
\end{equation*}
\begin{solution}
% sage: A = matrix(QQ, [[4,3,1], [2,0,5]])
% sage: A

% [2 0 5]
% sage: B = matrix(QQ, [[-1,0], [4,0], [3,5]])
% sage: B
% [-1  0]
% [ 4  0]
% [ 3  5]
% sage: A*B
% [11  5]
% [13 25]
\begin{equation*}
\begin{mat}
  4  &3  &1  \\

  3   &5
\end{mat}

13  &25 
\end{mat}
\end{equation*}
\end{solution}
\end{parts}
